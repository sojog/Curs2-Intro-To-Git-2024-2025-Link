## Setarea numelui global

# Setarea numelui utlizatorului
git config --global user.name "Numele tau..."
# Citirea numelui utlizatorului
git config --global user.name

# Setarea emailului utlizatorului
git config --global user.email "emailul@tau.com"
# Citirea emailului utlizatorului
git config --global user.email

## Deschiderea editorului Visual Studio Code ca editor default pentru git
git config --global core.editor "code --wait"

## Denumirea branch-ului initial main
git config --global init.defaultBranch main


## Initializeaza proiectul de git local - creaza un folder .git ascuns
git init

## Descarca de pe server un proiect de git
git clone https://github.com/username/project


## Arata lista de commituri 
git log
git log --oneline

## Arata statusul fisierelor (daca sunt schimbate)
git status 
git status -s 


### Comenzi de adaugarea a fisierelor in STAGED (inainte de creare a commitului)
git add filename
git add .
git add filename.*
git add filename1 filename2 filename3

### Comenzi de creare a commitului
git commit -m " Acesta este un mesaj descriptiv al commitului "
git commit --message " Acesta este un mesaj descriptiv al commitului "

### Comenda de adauga + commit  (!!! Fisierele trebuie sa fi fost incluse intr-un commit inainte)
git commit -am " Acesta este un mesaj descriptiv al commitului  "

## Modificarea mesajului ultimului commit 
git commit --amend

## Pentru a pune codul pe server
git push
git push origin

## Pentru a lua codul de pe server
git pull
git pull origin

## Pentru lucru cu branch-uri
## Listarea tuturor branch-urilor
git branch 

## Redenumirea branch-ului curent in ce este dupa -m
git branch -m main
